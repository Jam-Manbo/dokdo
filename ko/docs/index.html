# Dokdo를 프로젝트에 적용하기

몇 줄에 코드만 추가하면 됩니다.

const Dokdo = require('dokdo')

const DokdoHandler = new Dokdo(client, { prefix: '!' })

Dokdo 객채의 옵션을 더 확인해보고 싶으시다면 이곳을 참고해주세요.

메세지 핸들러:

client.on('message', async (message) => {
  await DokdoHandler.run(message)
  // 명령어 코드
})
